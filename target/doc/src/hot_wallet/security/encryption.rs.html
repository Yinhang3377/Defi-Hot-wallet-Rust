16/// WalletSecurity handles cryptographic operations for the hot wallet.
17/// This module provides secure encryption and decryption of sensitive data, such as private keys.
18pub struct WalletSecurity;
19
20impl WalletSecurity {
21    /// 使用 AES-256-GCM 加密私钥
22    ///
23    /// # 参数
24    /// * `private_key` - 待加密的私钥（如 secp256k1，32字节）
25    /// * `encryption_key` - 加密用的密钥（必须为64位十六进制字符串，解码后为32字节）
26    /// * `aad` - 关联数据（Associated Data），用于认证但不会被加密
27    ///
28    /// # 返回值
29    /// 返回包含随机nonce的密文（nonce前置），或 WalletError 错误
30    ///
31    /// # 安全说明
32    /// - 使用随机12字节nonce防止重放攻击
33    /// - nonce前置于密文，便于解密时提取
34    /// - 加密密钥用完自动清零，防止内存泄漏
35    /// - 生产环境建议用KDF安全派生encryption_key
36    pub fn encrypt_private_key(
37        private_key: &[u8],
38        encryption_key: &str,
39        aad: &[u8],
40    ) -> Result<Vec<u8>, WalletError> {
41        // 1. 验证私钥长度
42        if private_key.len() != 32 {
43            return Err(WalletError::EncryptionError("待加密的私钥长度必须为32字节".to_string()));
44        }
45
46        // 验证密钥是否为64位十六进制字符串，并解码为32字节
47        let key_bytes = hex::decode(encryption_key).map_err(|_| {
48            log::error!("加密密钥格式错误，无法从Hex解码: [已隐藏]");
49            WalletError::EncryptionError("加密密钥必须是有效的64位十六进制字符串".to_string())
50        })?;
51        if key_bytes.len() != 32 {
52            log::error!("加密密钥解码后长度不为32字节, 实际长度: [已隐藏]");
53            return Err(WalletError::EncryptionError("加密密钥解码后长度必须为32字节".to_string()));
54        }
55        if Self::is_weak_key(&key_bytes) {
56            log::error!("检测到弱加密密钥: [已隐藏]");
57            return Err(WalletError::EncryptionError(
58                "加密密钥过于简单，存在安全风险，请更换更复杂的密钥！".to_string(),
59            ));
60        }
61        // KDF: 用 Argon2 对明文密钥二次加固，salt 可用 AAD 或其它上下文
62        let salt = if !aad.is_empty() { aad } else { b"hotwallet-default-salt" };
63        let derived_key = Self::derive_key_from_env_key(&key_bytes, salt);
64
65        // 用派生密钥初始化加密器
66        let mut key = *Key::<Aes256Gcm>::from_slice(&derived_key);
67        let cipher = Aes256Gcm::new(&key);
68
69        // 生成随机12字节nonce
70        let nonce_bytes: [u8; 12] = rand::random();
71        let nonce = Nonce::from_slice(&nonce_bytes);
72
73        // 加密私钥
74        let ciphertext = cipher
75            .encrypt(nonce, Payload { msg: private_key, aad })
76            .map_err(|e| WalletError::EncryptionError(format!("加密失败: {}", e)))?;
77
78        // nonce前置于密文，便于解密
79        let mut result = Vec::with_capacity(nonce_bytes.len() + ciphertext.len());
80        result.extend_from_slice(&nonce_bytes);
81        result.extend_from_slice(&ciphertext);
82
83        // 用完自动清零密钥
84        // derived_key 会在作用域结束时自动清零（[u8; 32]）
85        key.zeroize();
86
87        Ok(result)
88    }
89
90    /// 解密用 AES-256-GCM 加密的私钥
91    ///
92    /// # 参数
93    /// * `ciphertext` - 加密数据（前12字节为nonce）
94    /// * `encryption_key` - 解密用密钥（必须为64位十六进制字符串）
95    /// * `aad` - 关联数据（Associated Data），必须与加密时使用的数据一致
96    ///
97    /// # 返回值
98    /// 返回解密后的私钥明文，或 WalletError 错误
99    ///
100    /// # 安全说明
101    /// - 默认前12字节为nonce
102    /// - 密钥用完自动清零
103    /// - 生产环境需校验密钥派生与nonce管理
104    #[allow(dead_code)]
105    pub fn decrypt_private_key(
106        ciphertext: &[u8],
107        encryption_key: &str,
108        aad: &[u8],
109    ) -> Result<Vec<u8>, WalletError> {
110        // 检查密文长度（必须大于12字节的nonce）
111        if ciphertext.len() <= 12 {
112            return Err(WalletError::EncryptionError("密文无效：长度不足，无有效数据".to_string()));
113        }
114
115        // 验证密钥是否为64位十六进制字符串，并解码为32字节
116        let mut key_bytes = hex::decode(encryption_key).map_err(|e| {
117            log::error!("解密密钥格式错误，无法从Hex解码: {}", e);
118            WalletError::EncryptionError("解密密钥必须是有效的64位十六进制字符串".to_string())
119        })?;
120
121        if key_bytes.len() != 32 {
122            log::error!("解密密钥解码后长度不为32字节, 实际长度: {}", key_bytes.len());
123            return Err(WalletError::EncryptionError("解密密钥解码后长度必须为32字节".to_string()));
124        }
125
126        // 提取nonce和实际密文
127        let nonce = Nonce::from_slice(&ciphertext[..12]);
128        let actual_ciphertext = &ciphertext[12..];
129
130        // 用密钥初始化解密器，这里的 from_slice 期望一个32字节的切片
131        let mut key = *Key::<Aes256Gcm>::from_slice(&key_bytes);
132        let cipher = Aes256Gcm::new(&key);
133
134        // 解密数据
135        let plaintext = cipher
136            .decrypt(nonce, Payload { msg: actual_ciphertext, aad })
137            .map_err(|e| WalletError::EncryptionError(format!("解密失败: {}", e)))?;
138
139        // 用完自动清零密钥
140        key_bytes.zeroize();
141        // 深度防御：显式清零 Key 对象本身
142        key.zeroize();
143
144        Ok(plaintext)
145    }
146    // ---- 内部辅助函数 ----
147    fn is_weak_key(key_bytes: &[u8]) -> bool {
148        if key_bytes.iter().all(|&b| b == 0) || key_bytes.iter().all(|&b| b == 0xff) {
149            return true;
150        }
151        if key_bytes.len() >= 2 {
152            let first = key_bytes[0];
153            let second = key_bytes[1];
154            if key_bytes
155                .iter()
156                .enumerate()
157                .all(|(i, &b)| if i % 2 == 0 { b == first } else { b == second })
158            {
159                return true;
160            }
161        }
162        false
163    }
164    fn derive_key_from_env_key(env_key: &[u8], salt: &[u8]) -> [u8; 32] {
165        let mut out = [0u8; 32];
166        for (i, b) in out.iter_mut().enumerate() {
167            *b = env_key.get(i % env_key.len()).cloned().unwrap_or(0)
168                ^ salt.get(i % salt.len()).cloned().unwrap_or(0);
169        }
170        out
171    }
172}

178    #[test]
179    fn test_encrypt_decrypt_cycle() {
180        // 测试加密解密流程
181        let private_key = b"32_byte_private_key_1234567890ab";
182        // 64位十六进制字符串
183        let encryption_key = "33325f627974655f656e6372797074696f6e5f6b65795f31323334353637383930";
184        let aad = b"associated_data";
185
186        let encrypted =
187            WalletSecurity::encrypt_private_key(private_key, encryption_key, aad).unwrap();
188        let decrypted =
189            WalletSecurity::decrypt_private_key(&encrypted, encryption_key, aad).unwrap();
190        assert_eq!(private_key.to_vec(), decrypted);
191    }

193    #[test]
194    fn test_invalid_key_length() {
195        // 测试密钥长度不足的情况
196        let private_key = b"test_key";
197        let aad = b"";
198        let short_key = "too_short";
199        // 私钥长度不足32字节，应该报错
200        let result = WalletSecurity::encrypt_private_key(private_key, short_key, aad);
201        assert!(matches!(result, Err(WalletError::EncryptionError(_))));
202
203        // 密钥格式错误，应该报错
204        let valid_private_key = &[0; 32];
205        let result2 = WalletSecurity::encrypt_private_key(valid_private_key, short_key, aad);
206        assert!(matches!(result2, Err(WalletError::EncryptionError(_))));
207    }
